SRA(noisy_vol, N_iters):
  acc = copy(noisy_vol)
  for i in range(N_iters):
    denoised_vol = acc/(i + 1)
    shuffled_vol = shuffle(noisy_vol)
    acc += project(denoised_vol, shuffled_vol)
  denoised_vol = acc/(N_iters + 1)
  return denoised
